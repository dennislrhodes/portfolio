What is the arrest rate for domestic battery? 

%%sql
    SELECT (arrested_domestic_battery::float/total_domestic_battery::float)::float as Domestic_Battery_Rate
    FROM(
    SELECT arrested_domestic_battery, total_domestic_battery, ROW_NUMBER() OVER (PARTITION BY arrested_domestic_battery)
    FROM ( select *
    FROM (SELECT COUNT(*) as arrested_domestic_battery
    FROM crime
    JOIN crime_report
    USING (crime_id)
    WHERE domestic = True AND arrest = True AND primary_desc LIKE '%BATTERY%') as s1
    ) as s2
    NATURAL JOIN (SELECT total_domestic_battery, ROW_NUMBER() OVER (PARTITION BY total_domestic_battery)
    FROM ( select *
    FROM (SELECT COUNT(*) as total_domestic_battery
    FROM crime
    JOIN crime_report
    USING (crime_id)
    WHERE domestic = True AND primary_desc LIKE '%BATTERY%') as s1
    ) as s2
    ) as s3 ) as s4
    ;
